    for n := sl.level - 1; n >= 0; n-- {
        for cur.level[n] != nil {
            if cur.level[n].score == score {
                return true

func (sl *zskiplist) Add(score int, value interface{}) {
    // update_array
    update := make([]*zskiplistNode, sl.level)

    // update
    level := randLevel()
    if level > sl.level {
        for i := sl.level; i < level; i++ {
            update[i] = sl.head
        }
        sl.level = level
    }
    node := newNode(score, value, level)
    for i := 0; i < level; i++ {
        node.level[i] = update[i].level[i]
        update[i].level[i] = node
    }
}
