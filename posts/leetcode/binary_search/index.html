func binSearch(arr []int, k int) int {
    left, right := 0, len(arr) - 1
    for left <= right {
        mid := (left+right) / 2
        if arr[mid] == k{
            return mid
        } else if arr[mid] < k {
            left = mid + 1
        } else {
            right = mid - 1
        }
    }
    return -1
}

func bsFirst(arr []int, k int) {
    left, right := 0, len(arr) - 1
    for left < right {
        mid := (left+right)/2
        if arr[mid] < k {
            left = mid + 1 
        }else {
            right = mid
        }
    }
    return left
}
func bsLast(arr []int, k int) {
    left, right := 0, len(arr) - 1
    for left < right {
        mid := (left+right)/2
        if arr[mid] > k {
            right = mid - 1
        }else {
            left = mid
        }
    }
    return left
}
