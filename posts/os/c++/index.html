// lock free stack
struct Node {
	int value;
	Node *next;
}
struct Stack {
	Node *top;
} 
void init(Stack *s) {
	s->top = NULL;
}
void push(Stack *s, Node *n) {
	while(1) {
		Node *old_top = s->top;
		n->next = old_top;
		if (compare_and_swap(&s->top, old_top, n) == old_top) {
			return;
		}
	}
}
Node* pop(Stack *s) {
	while(1) {
		Node *old_top = s->top;
		if old_top == NULL { return NULL;}
		Node *new_top = old_top->next;
		if (compare_and_swap(&s->top, old_top, new_top)==old_top) {
			return old_top;
		}
	}
}

struct Stack {
	Node* top;
	int pop_count;

Node* pop(Stack* s) {
	while (1) {
		int pop_count = s->pop_count;
		Node* top = s->top;
		if (top == NULL)
			return NULL;
		Node* new_top = top->next;
		if (double_compare_and_swap(&s->top, top, new_top,
			&s->pop_count, pop_count, pop_count+1))
			return top;
	}
}
