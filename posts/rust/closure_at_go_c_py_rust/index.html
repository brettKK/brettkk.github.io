func toNFA(postfixExp []byte) *NFA {
    if len(postfixExp) == 0 { return fromEpsilon()}
    stack := make([]*NFA, 0)
    for _, token := range postfixExp {
        switch token {
        case '*':
            nfa := stack[len(stack) - 1]   // peek the top on the stack
            stack = stack[ :len(stack) - 1] // pop the stack
            stack = append(stack, closure(nfa))
        case '|':
            rightNFA := stack[len(stack) - 1]
            stack = stack[ :len(stack) - 1]
            leftNFA := stack[len(stack) - 1]
            stack = stack[:len(stack) - 1]
            stack = append(stack, union(leftNFA, rightNFA)) //
        case '.':
            rightNFA := stack[len(stack) - 1]
            stack = stack[ :len(stack) - 1]
            leftNFA := stack[len(stack) - 1]
            stack = stack[:len(stack) - 1]
            stack = append(stack, concat(leftNFA, rightNFA))
        default:
            stack = append(stack, fromSymbol(token))
        }
    }
    return stack[len(stack) - 1] // æ ˆé¡¶ä¸ºæœ€ç»ˆçš„NFA
}
