trait Iterator<T> {
    fn next(&mut self) -> Option<T>;
}
struct container<T> {
    data: Vec<T>,
}
struct containerIter<'a, T> {
    index: usize,
    container: &'a container<T,
}
impl container {
    fn iter(&self) -> containerIter {
        containerIter{
            index: 0,
            container: &self,
        }
    } 
}

impl<T> Iterator<T> for containerIter{
    fn next(&mut self) -> Option<T> {
        let item = self.container.data.get(self.index).cloned();
        self.index += 1;
        item
    }
}
